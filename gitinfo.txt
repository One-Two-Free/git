git diff	показать изменения которые не индексировали
    --staged	сравнивает индексированные изменения и последнеи фиксированные версии
    		что из проиндексированного войдет в коммит
    --check	ошибки с пробелами
commit -a -m 	автоматически индексирует отслеживаемы файлы перед фиксацией

git rm 	  	удалить файл из числа отслеживаемых, индексирует удаление файла
git rm --cashed	удалить из области индексации, оставить в рабочей папке
git mv oldFile newFile	переименовать файл

log -p -2      	разница внесенная каждым коммитом, -2 последние две записи
    --stat	краткая статистика
    --pretty=oneline	еще параметры short, full, fuller	format стр50
    --pretty=format:"%h - %an, %ar : %s"

log --graph	граф с историей ветвлений и слияний
    --decorate	куда нацелены указатели веток

commit -amend	повторное сохранение версии, область индексирования включает в коммит

reset HEAD <имя файла> отмена индексирования файла


УДАЛЕННЫЙ РЕПОЗИТОРИЙ
remote	  список коротких имен удаленных репозиторий
remote -v список url
remote add [name] [url]	добавить удаленный репозиторий
fetch name  скачать с удаленного репозитория, не выполняет автоматическое слияние
remote rm удалить сылку на удаленный репозиторий

push [удаленный серве] [ветка]
pull 		извлекает информацию из удаленной ветки и слияние с текущей веткой
remote show [удаленный сервер] информация об удаленном репозитории



branch 	    новвая ветка
branch -d   удалить ветку
checkout    перейти на ветку
merge	    слияние
checkout -b [ветка] [удаленный сервер]/[ветка]  копия удаленной ветки
checkout -b --track [удаленный сервер]/[ветка]	аналог предыдущей операции

branch -u origin/serverfix     текущей ветке сопоставить удаленную ветку, -u или --set-upstream-to
branch -vv		       список веток наблюдения за удаленными ветками, на сколько ветка опережает удаленную,
       			       все цифры локальные данные, зафиксированы в момент последнего скачивания
для просмотра актуальных данных
fetch --all   извлечь данные со всех удаленных серверов
branch -vv

push [удаленный сервер] --delete [ветка]	удалить удаленную ветку


rebase master	перетащить измениня из текущей ветки в ветку мастер
